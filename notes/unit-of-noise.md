$$
\begin{align}
\int \delta(\tau) d\tau &= 1 \\
[\tau] &= \mathrm{sec} \\
[\delta(\tau)] &= \frac{1}{\mathrm{sec}} = \mathrm{Hz}
\end{align}
$$


$$
\begin{align}

\end{align}
$$

$$
\begin{align}{}
[\omega] = [b_g] = [\varepsilon_g] &= \frac{\mathrm{rad}}{\mathrm{sec}} \\
[\dot{b}_g] = [\eta_g] &= \frac{\mathrm{rad}}{\mathrm{sec}^2}
\end{align}
$$

$$
\begin{align}{}
